You can derive from Builder and extend it with your own class members. Methods such as AsGenerator, [ToEnumerable](xref: Sharpy.BuilderExtensions.ToEnumerable2(0,System.Func{0,System.Int32,1},System.Int32)) and ToGenerator will have access to the descended class members as you will see in the following examples.

Calling AsGenerator

using Sharpy.Core;
using Sharpy.Enums;
using Sharpy;
using System.Generic.Collections;
using System;
namespace Example
{
    class Program
    {
        public static void Main (string[] args)
        {
            // Creates a IGenerator<string> which will invoke MyMethod.
            IGenerator<string> gen = ExampleClass.AsGenerator(TBuilder: new ExampleClass() , selector: (ExampleClass: e) => e.MyMethod()):
        }
    }

    // Example class deriving from Builder.
    class ExampleClass : Builder
    {
        public string MyMethod()
        {
            return "something to return";
        }
    }
}

using Sharpy.Core;
using Sharpy.Enums;
using Sharpy;
using System.Generic.Collections;
using System;
namespace Example
{
    class Program
    {
        public static void Main (string[] args)
        {
            ExampleClass example = new ExampleClass();
            // Creates an IEnumerable<string> with 20 elements.
            IEnumerable<string> gen = example.ToEnumerable(selector: (ExampleClass e) => e.MyMethod(), count: 20);

        }
    }

    // Example class deriving from Builder.
    class ExampleClass : Builder
    {
        public string MyMethod()
        {
            return "something to return";
        }
    }
}
